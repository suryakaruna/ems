CREATE TABLE user(
	id int not null AUTO_INCREMENT,
    username varchar(70),
    email varchar(50),
    pass varchar(35),
    phone varchar(15),
    userType varchar(5),
	submit timestamp,
    PRIMARY KEY(id)
)


CREATE TABLE event(
	id int not null AUTO_INCREMENT,
    title varchar(70),
    type varchar(100),
    category varchar(150),
    venue varchar(300),
    online BINARY,
    banner varchar(200),
    description varchar(500), 
	startT timestamp,
    endT timestamp,
    submit timestamp,
    PRIMARY KEY(id)
)
